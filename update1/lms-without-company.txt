CREATE TABLE IF NOT EXISTS public.users
(
    id integer NOT NULL GENERATED ALWAYS AS IDENTITY ( INCREMENT 1 START 1 MINVALUE 1 MAXVALUE 2147483647 CACHE 1 ),
    name character varying(250) COLLATE pg_catalog."default",
    email character varying(250) COLLATE pg_catalog."default",
    gender character varying(250) COLLATE pg_catalog."default",
    mobile character varying(50) COLLATE pg_catalog."default",
    password character varying(250) COLLATE pg_catalog."default",
    isadmin text COLLATE pg_catalog."default",
    issuperadmin character varying(10) COLLATE pg_catalog."default",
    image bytea,
    CONSTRAINT users_pkey PRIMARY KEY (id)
)



CREATE TABLE IF NOT EXISTS public.leads
(
    id integer NOT NULL,
    name character varying(250) COLLATE pg_catalog."default",
    email character varying(250) COLLATE pg_catalog."default",
    address character varying(300) COLLATE pg_catalog."default",
    mobile character varying(50) COLLATE pg_catalog."default",
    source character varying(100) COLLATE pg_catalog."default",
    date character varying(100) COLLATE pg_catalog."default",
    "time" character varying(100) COLLATE pg_catalog."default",
    owner character varying(200) COLLATE pg_catalog."default",
    currentowner character varying(200) COLLATE pg_catalog."default",
    status character varying(100) COLLATE pg_catalog."default",
    priority character varying(50) COLLATE pg_catalog."default",
    CONSTRAINT leads_pkey PRIMARY KEY (id)
)

CREATE TABLE IF NOT EXISTS public.contacts
(
    id integer,
    name character varying(250) COLLATE pg_catalog."default",
    email character varying(250) COLLATE pg_catalog."default",
    mobile character varying(250) COLLATE pg_catalog."default",
    comments character varying(250) COLLATE pg_catalog."default"
)

CREATE TABLE IF NOT EXISTS public.comments
(
    id integer NOT NULL GENERATED ALWAYS AS IDENTITY ( INCREMENT 1 START 1 MINVALUE 1 MAXVALUE 2147483647 CACHE 1 ),
    leadid integer,
    userid integer,
    useremail character varying COLLATE pg_catalog."default",
    comment character varying COLLATE pg_catalog."default",
    creation_date timestamp without time zone DEFAULT CURRENT_TIMESTAMP,
    updatation_date timestamp without time zone,
    CONSTRAINT comments_pkey PRIMARY KEY (id)
)
